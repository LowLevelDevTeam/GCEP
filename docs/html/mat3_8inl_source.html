99 template<typename T>
100 Mat3<T> Mat3<T>::Inverted() const
101 {
102 T det = Determinant();
103
104 // If the determinant is zero, the matrix is not invertible.
105 // Return the identity matrix instead.
106 if (std::abs(det) < static_cast<T>(1e-6))
107 return Mat3<T>::Identity();
108
109 T invDet = static_cast<T>(1) / det;
110
111 return Mat3<T>(
112 (m[1][1] * m[2][2] - m[1][2] * m[2][1]) * invDet,
113 (m[0][2] * m[2][1] - m[0][1] * m[2][2]) * invDet,
114 (m[0][1] * m[1][2] - m[0][2] * m[1][1]) * invDet,
115
116 (m[1][2] * m[2][0] - m[1][0] * m[2][2]) * invDet,
117 (m[0][0] * m[2][2] - m[0][2] * m[2][0]) * invDet,
118 (m[0][2] * m[1][0] - m[0][0] * m[1][2]) * invDet,
119
120 (m[1][0] * m[2][1] - m[1][1] * m[2][0]) * invDet,
121 (m[0][1] * m[2][0] - m[0][0] * m[2][1]) * invDet,
122 (m[0][0] * m[1][1] - m[0][1] * m[1][0]) * invDet
123 );
124 }
