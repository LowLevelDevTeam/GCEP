19// Log level — strongly typed, bitmask-capable
20enum class LogLevel : uint32_t
21{
22 eNone = 0,
23 eCommon = 1 << 0,
24 eWarn = 1 << 1,
25 eError = 1 << 2,
26 eVerbose = 1 << 3,
27 eAll = eCommon | eWarn | eError | eVerbose
28};

30inline LogLevel operator|(LogLevel a, LogLevel b)
31{
32 return static_cast<LogLevel>(static_cast<uint32_t>(a) | static_cast<uint32_t>(b));
33}
34inline LogLevel operator&(LogLevel a, LogLevel b)
35{
36 return static_cast<LogLevel>(static_cast<uint32_t>(a) & static_cast<uint32_t>(b));
37}
38inline bool hasFlag(LogLevel mask, LogLevel flag)
