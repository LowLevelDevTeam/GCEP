92 {

95
96 // Avoid division by zero
97 if (magnitudeProduct == static_cast<T>(0))
98 return static_cast<T>(0);
99
100 T cosTheta = dot / magnitudeProduct;
101
102 // Clamp the value to avoid domain errors with acos()
103 if (cosTheta > static_cast<T>(1))

105 else if (cosTheta < static_cast<T>(-1))

109 }
