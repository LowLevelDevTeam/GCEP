148 struct ShapeKey
149 {
150 EPhysicsShapeType type = EPhysicsShapeType::BOX;
151 Vector3<float> scale{};
152 float radius = 0.f;
153 float height = 0.f;
154
155 bool operator==(const ShapeKey& other) const
156 {
157 return type == other.type &&
158 scale == other.scale &&
159 radius == other.radius &&
160 height == other.height;
161 }
162 };
163
164 struct ShapeKeyHasher
165 {
166 size_t operator()(const ShapeKey& key) const
167 {
168 size_t h1 = std::hash<int>()((int)key.type);
169 size_t h2 = std::hash<float>()(key.scale.x);
170 size_t h3 = std::hash<float>()(key.scale.y);
171 size_t h4 = std::hash<float>()(key.scale.z);
172 size_t h5 = std::hash<float>()(key.radius);
173 size_t h6 = std::hash<float>()(key.height);
174
175 return h1 ^ h2 ^ h3 ^ h4 ^ h5 ^ h6;
176 }
