133 template <typename T>
134 Mat4<T> Mat4<T>::Rotate(const Quaternion& rotation)
135 {
136 Mat4<T> result = Identity();
137 T xx = static_cast<T>(rotation.x) * static_cast<T>(rotation.x);
138 T yy = static_cast<T>(rotation.y) * static_cast<T>(rotation.y);
139 T zz = static_cast<T>(rotation.z) * static_cast<T>(rotation.z);
140 T xy = static_cast<T>(rotation.x) * static_cast<T>(rotation.y);
141 T xz = static_cast<T>(rotation.x) * static_cast<T>(rotation.z);
142 T yz = static_cast<T>(rotation.y) * static_cast<T>(rotation.z);
143 T wx = static_cast<T>(rotation.w) * static_cast<T>(rotation.x);
144 T wy = static_cast<T>(rotation.w) * static_cast<T>(rotation.y);
145 T wz = static_cast<T>(rotation.w) * static_cast<T>(rotation.z);
146
147 result.m[0][0] = static_cast<T>(1) - static_cast<T>(2) * (yy + zz);
148 result.m[0][1] = static_cast<T>(2) * (xy + wz);
149 result.m[0][2] = static_cast<T>(2) * (xz - wy);
150 result.m[1][0] = static_cast<T>(2) * (xy - wz);
151 result.m[1][1] = static_cast<T>(1) - static_cast<T>(2) * (xx + zz);
152 result.m[1][2] = static_cast<T>(2) * (yz + wx);
153 result.m[2][0] = static_cast<T>(2) * (xz + wy);
154 result.m[2][1] = static_cast<T>(2) * (yz - wx);
155 result.m[2][2] = static_cast<T>(1) - static_cast<T>(2) * (xx + yy);
156 return result;
157 }

416 template <typename T>
417 Quaternion Mat4<T>::ExtractRotation() const
418 {
419 // Convert upper-left 3x3 to quaternion using standard matrix-to-quaternion conversion.
420 // Assumes matrix is rotation (possibly with scale). We remove scale first.
421 Vector3<T> scale = ExtractScale();
422 T sx = scale.x, sy = scale.y, sz = scale.z;
423 T invSx = (sx != static_cast<T>(0)) ? static_cast<T>(1) / sx : static_cast<T>(1);
424 T invSy = (sy != static_cast<T>(0)) ? static_cast<T>(1) / sy : static_cast<T>(1);
425 T invSz = (sz != static_cast<T>(0)) ? static_cast<T>(1) / sz : static_cast<T>(1);
426
427 T r00 = m[0][0] * invSx;
428 T r01 = m[0][1] * invSy;
429 T r02 = m[0][2] * invSz;
430 T r10 = m[1][0] * invSx;
431 T r11 = m[1][1] * invSy;
432 T r12 = m[1][2] * invSz;
433 T r20 = m[2][0] * invSx;
434 T r21 = m[2][1] * invSy;
435 T r22 = m[2][2] * invSz;
436
437 T trace = r00 + r11 + r22;
438 T qw, qx, qy, qz;
439
440 if (trace > static_cast<T>(0))
441 {
442 T s = std::sqrt(static_cast<double>(trace) + 1.0) * static_cast<T>(2);
443 qw = static_cast<T>(0.25) * s;
444 qx = (r21 - r12) / s;
445 qy = (r02 - r20) / s;
446 qz = (r10 - r01) / s;
447 }
448 else if ((r00 > r11) && (r00 > r22))
449 {
450 T s = std::sqrt(static_cast<double>(1.0 + r00 - r11 - r22)) * static_cast<T>(2);
451 qw = (r21 - r12) / s;
452 qx = static_cast<T>(0.25) * s;
453 qy = (r01 + r10) / s;
454 qz = (r02 + r20) / s;
455 }
456 else if (r11 > r22)
457 {
458 T s = std::sqrt(static_cast<double>(1.0 + r11 - r00 - r22)) * static_cast<T>(2);
459 qw = (r02 - r20) / s;
460 qx = (r01 + r10) / s;
461 qy = static_cast<T>(0.25) * s;
462 qz = (r12 + r21) / s;
463 }
464 else
465 {
466 T s = std::sqrt(static_cast<double>(1.0 + r22 - r00 - r11)) * static_cast<T>(2);
467 qw = (r10 - r01) / s;
468 qx = (r02 + r20) / s;
469 qy = (r12 + r21) / s;
470 qz = static_cast<T>(0.25) * s;
471 }
472
473 // Quaternion class uses floats; cast safely.
474 return Quaternion(static_cast<float>(qw), static_cast<float>(qx), static_cast<float>(qy), static_cast<float>(qz));
475 }

477 template <typename T>
478 bool Mat4<T>::ValidTRS() const
479 {
480 // Check last row is (0,0,0,1) and upper-left 3x3 is orthonormal up to scale.
481 const double EPS = 1e-5;
482 if (std::fabs(static_cast<double>(m[0][3])) > EPS) return false;
483 if (std::fabs(static_cast<double>(m[1][3])) > EPS) return false;
484 if (std::fabs(static_cast<double>(m[2][3])) > EPS) return false;
485 if (std::fabs(static_cast<double>(m[3][3] - static_cast<T>(1))) > EPS) return false;
486
487 Vector3<T> col0 = GetColumn(0);
488 Vector3<T> col1 = GetColumn(1);
489 Vector3<T> col2 = GetColumn(2);
490
491 // Non-zero scale?
492 Vector3<T> scale = ExtractScale();
493 if (scale.x == static_cast<T>(0) || scale.y == static_cast<T>(0) || scale.z == static_cast<T>(0))
494 return false;
495
496 // Normalize columns by scale and check orthonormality
497 Vector3<T> n0 = Vector3<T>(col0.x / scale.x, col0.y / scale.x, col0.z / scale.x).Normalized();
498 Vector3<T> n1 = Vector3<T>(col1.x / scale.y, col1.y / scale.y, col1.z / scale.y).Normalized();
499 Vector3<T> n2 = Vector3<T>(col2.x / scale.z, col2.y / scale.z, col2.z / scale.z).Normalized();
500
501 double d01 = std::fabs(static_cast<double>(n0.Dot(n1)));
502 double d02 = std::fabs(static_cast<double>(n0.Dot(n2)));
503 double d12 = std::fabs(static_cast<double>(n1.Dot(n2)));
504
505 if (d01 > EPS || d02 > EPS || d12 > EPS) return false;
506
507 return true;
508 }
