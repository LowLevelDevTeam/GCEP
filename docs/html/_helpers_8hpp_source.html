163inline std::string getFileIcon(const std::filesystem::path& file)
164{
165 const std::string ext = file.extension().string();
166
167 if (ext == ".png" || ext == ".jpg" || ext == ".jpeg") return ICON_FA_FILE_IMAGE_O;
168 if (ext == ".mp3" || ext == ".wav") return ICON_FA_FILE_AUDIO_O;
169 if (ext == ".mp4" || ext == ".avi") return ICON_FA_FILE_VIDEO_O;
170 if (ext == ".cpp" || ext == ".hpp" || ext == ".h") return ICON_FA_FILE_CODE_O;
171 if (ext == ".zip" || ext == ".rar") return ICON_FA_FILE_ARCHIVE_O;
172
173 return ICON_FA_FILE;
174}
