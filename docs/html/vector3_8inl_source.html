215 T dot = a.Dot(b);
216 T mag = a.Length() * b.Length();
217 if (mag == static_cast<T>(0))
218 return static_cast<T>(0);
219
220 T cosTheta = dot / mag;
221 cosTheta = max(static_cast<T>(-1), min(static_cast<T>(1), cosTheta));
222 return std::acos(cosTheta) * static_cast<T>(180.0 / std::numbers::pi_v<float>);
