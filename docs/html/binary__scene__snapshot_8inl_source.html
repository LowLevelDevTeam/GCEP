6inline void binarySnapShot::serializeAll(SER::FileArchive& archive)
7{
8 const auto& registry = m_scene.getRegistry();
9 const auto& pools = registry.getPools();
10
11 archive.writeName("GCEP_SCENE");
12 archive.writeUint64(1); // version
13 archive.writeName(m_scene.getName());
14
15 uint32_t poolCount = 0;
16 for (auto& pool : pools)
17 if (pool && !pool->getEntities().empty()) poolCount++;
18
19 archive.writeSize(poolCount);
20
21 for (auto& pool : pools)
22 {
23 if (!pool) continue;
24 const auto& entities = pool->getEntities();
25 if (entities.empty()) continue;
26
27 archive.writeName(pool->getName());
28 archive.writeSize(static_cast<uint32_t>(entities.size()));
29
30 for (const auto& entity : entities)
31 {
32 archive.writeUint32(entity & 0xFFFFFF);
33
34 std::streampos sizePos = archive.tell();
35 archive.writeUint32(0); // placeholder byte-size
36
37 std::streampos dataStart = archive.tell();
38 pool->serializeEntity(entity, archive);
39 std::streampos dataEnd = archive.tell();
40
41 uint32_t byteCount = static_cast<uint32_t>(dataEnd - dataStart);
42 archive.seek(sizePos);
43 archive.writeUint32(byteCount);
44 archive.seek(dataEnd);
45
46 Log::info(std::format("Pool Name : {} | Entity : {} | Saved" , pool->getName(), entity));
47 }
48 }
