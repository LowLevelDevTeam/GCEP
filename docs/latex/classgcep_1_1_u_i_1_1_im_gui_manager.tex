\doxysubsection{Detailed Description}
RAII manager for the Dear Im\+Gui context and its Vulkan + GLFW backends. 

{\ttfamily \doxylink{classgcep_1_1_u_i_1_1_im_gui_manager}{Im\+Gui\+Manager}} centralises the three-\/step Im\+Gui lifecycle — initialisation, per-\/frame begin/\+end, and shutdown — so that the rest of the editor code only calls the high-\/level helpers and never touches the raw backend functions directly.

\begin{DoxyParagraph}{Typical usage}

\begin{DoxyCode}{0}
\DoxyCodeLine{UI::ImGuiManager\ imguiManager;}
\DoxyCodeLine{imguiManager.init(glfwWindow,\ vulkanInitInfo);}
\DoxyCodeLine{}
\DoxyCodeLine{\textcolor{keywordflow}{while}\ (running)}
\DoxyCodeLine{\{}
\DoxyCodeLine{\ \ \ \ imguiManager.beginFrame();}
\DoxyCodeLine{\ \ \ \ \textcolor{comment}{//\ ...\ ImGui\ calls\ ...}}
\DoxyCodeLine{\ \ \ \ imguiManager.endFrame();}
\DoxyCodeLine{\}}
\DoxyCodeLine{}
\DoxyCodeLine{imguiManager.shutdown();}

\end{DoxyCode}

\end{DoxyParagraph}
\begin{DoxyNote}{Note}
All methods must be called from the main/\+render thread. 
\end{DoxyNote}
