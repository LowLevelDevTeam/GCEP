235 inline std::string getFileIcon(const std::filesystem::path& file)
236 {
237 const std::string ext = file.extension().string();
238
239 if (ext == ".png" || ext == ".jpg" || ext == ".jpeg") return ICON_FA_FILE_IMAGE_O;
240 if (ext == ".mp3" || ext == ".wav") return ICON_FA_FILE_AUDIO_O;
241 if (ext == ".mp4" || ext == ".avi") return ICON_FA_FILE_VIDEO_O;
242 if (ext == ".cpp" || ext == ".hpp" || ext == ".h") return ICON_FA_FILE_CODE_O;
243 if (ext == ".zip" || ext == ".rar") return ICON_FA_FILE_ARCHIVE_O;
244 if (ext == ".gcprefab") return ICON_FA_ARCHIVE;
245
246 return ICON_FA_FILE;
247 }
